/**
 * @description Tests for initDevices function in server.js
 * @module initDevices
 */
const { initDevices } = require('../../../server');
const Device = require('../../../models/Device');
const mongoose = require('mongoose');
const path = require('path');
const xlsx = require('xlsx');
const fs = require('fs');

describe('initDevices', () => {
  beforeEach(async () => {
    await Device.deleteMany();
  });

  /**
   * @description Test case for detecting invalid devices in the database
   * @test {initDevices}
   */
  it('should throw error if invalid devices are found', async () => {
    // 잘못된 데이터 삽입
    await Device.create({
      osName: 'AOS' // serialNumber 누락
    });

    await expect(initDevices()).rejects.toThrow('Invalid devices found');
  });

  /**
   * @description Test case for successful device import from Excel file
   * @test {initDevices}
   */
  it('should import devices successfully from Excel file', async () => {
    const exportPath = path.join(__dirname, 'test.xlsx');
    // 테스트용 엑셀 파일 생성
    const wb = xlsx.utils.book_new();
    const ws = xlsx.utils.json_to_sheet([
      { '시리얼 번호': 'TEST001', 'OS 이름': 'AOS', 'OS 버전': '14', '모델명': 'TestDevice' }
    ]);
    xlsx.utils.book_append_sheet(wb, ws, 'Devices');
    xlsx.writeFile(wb, exportPath);

    await initDevices(false, exportPath);
    const devices = await Device.find();
    expect(devices.length).toBe(1);
    expect(devices[0].serialNumber).toBe('TEST001');
    expect(devices[0].osName).toBe('AOS');

    // 테스트 파일 삭제
    fs.unlinkSync(exportPath);
  });

  /**
   * @description Test case for detecting invalid data in Excel file
   * @test {initDevices}
   */
  it('should throw error if Excel file contains invalid data', async () => {
    const exportPath = path.join(__dirname, 'test.xlsx');
    // 잘못된 데이터 포함 엑셀 파일 생성
    const wb = xlsx.utils.book_new();
    const ws = xlsx.utils.json_to_sheet([
      { '시리얼 번호': 'TEST001', 'OS 이름': 'AOS' }, // 정상 데이터
      { '시리얼 번호': '', 'OS 이름': 'AOS' }, // 필수 필드 누락
      { '시리얼 번호': 'TEST003', 'OS 이름': 'AOS', '대여일시': 'invalid-date' }, // 형식 오류
      { '시리얼 번호': 'TEST001', 'OS 이름': 'AOS' }, // 중복 데이터
      { '시리얼 번호': 'TEST004', 'OS 이름': 'AOS', 'location': 'OldField' } // 과거 필드
    ]);
    xlsx.utils.book_append_sheet(wb, ws, 'Devices');
    xlsx.writeFile(wb, exportPath);

    await expect(initDevices(false, exportPath)).rejects.toThrow('Invalid devices found');

    // 테스트 파일 삭제
    fs.unlinkSync(exportPath);
  });
});
